// ======================================================== // شات حسوني - النسخة الإمبراطورية (الكود الكامل والمعدل) // ======================================================== // 1. الإعدادات الأساسية وتثبيت الروم العام الافتراضي لجميع الأعضاء const DEFAULT_ROOM = "general_room"; let currentUserId = ""; // يتم تعيينه عند تسجيل دخول المستخدم let currentUsername = ""; // يتم تعيينه عند تسجيل دخول المستخدم // دالة دخول أي عضو جديد أو موجود وتوجيهه تلقائياً للروم العام ليظهر للكل function joinGeneralRoom(userId, username) { currentUserId = userId; currentUsername = username; // تسجيل المستخدم في قائمة المتواجدين داخل الروم العام فوراً في قاعدة البيانات db.collection("rooms").doc(DEFAULT_ROOM).collection("users").doc(userId).set({ username: username, status: "online", joinedAt: firebase.firestore.FieldValue.serverTimestamp() }).then(() => { console.log("تم دخول العضو بنجاح في الروم العام للجميع."); }).catch((error) => { console.error("خطأ في تسجيل الدخول للروم العام:", error); }); } // 2. الاستماع اللحظي المستمر (Listeners) - لضمان عدم فصل الرسائل وظهورها للجميع دائماً function listenToChatUpdates() { // أولاً: مراقبة إعدادات الروم (إذا قام الملك بتعديل الخلفية أو أي شيء يظهر للجميع فوراً) db.collection("rooms").doc(DEFAULT_ROOM).onSnapshot((doc) => { if (doc.exists()) { const roomData = doc.data(); // تغيير خلفية الروم تلقائياً عند جميع الأعضاء المتواجدين فور تعديلها من المالك if (roomData.background) { document.body.style.backgroundImage = `url('${roomData.background}')`; document.body.style.backgroundSize = "cover"; document.body.style.backgroundPosition = "center"; document.body.style.backgroundAttachment = "fixed"; } } }, (error) => { console.error("خطأ في استقبال تحديثات الغرفة من قاعدة البيانات:", error); }); // ثانياً: مراقبة واستقبال الرسائل في الروم العام (تظهر للكل دائمًا بدون انقطاع) db.collection("rooms").doc(DEFAULT_ROOM).collection("messages") .orderBy("timestamp", "asc") .onSnapshot((snapshot) => { // تحديد حاوية الرسائل في الـ HTML لديك const messagesContainer = document.getElementById("messages-container"); if (!messagesContainer) return; // تفريغ الحاوية وإعادة بنائها بدقة لضمان التزامن الكامل ومنع التكرار messagesContainer.innerHTML = ""; snapshot.forEach((doc) => { const msg = doc.data(); // هيكل عرض الرسالة (يمكنك تعديل الكلاسات حسب تصميمك الفخم) const messageHtml = `
${msg.sender}: ${msg.text}
`; messagesContainer.innerHTML += messageHtml; }); // النزول التلقائي لأسفل الشات مع كل رسالة جديدة للجميع messagesContainer.scrollTop = messagesContainer.scrollHeight; }, (error) => { console.error("خطأ في مزامنة الرسائل اللحظية:", error); }); // ثالثاً: مراقبة إنشاء غرف جديدة لتظهر للجميع فوراً في قائمة الغرف عند إنشائها db.collection("rooms").onSnapshot((snapshot) => { const roomsListContainer = document.getElementById("rooms-list"); if (!roomsListContainer) return; snapshot.docChanges().forEach((change) => { if (change.type === "added") { const room = change.doc.data(); const roomId = change.doc.id; // إظهار الغرفة الفرعية للجميع في القائمة بشرط ألا تكون الروم العام لمنع التكرار if (roomId !== DEFAULT_ROOM && !document.getElementById(roomId)) { const roomElement = `
${room.name}
`; roomsListContainer.innerHTML += roomElement; } } }); }); } // 3. دالة إرسال الرسائل من قبل الأعضاء إلى الروم العام function sendMessage(textMessage) { if (textMessage.trim() === "") return; db.collection("rooms").doc(DEFAULT_ROOM).collection("messages").add({ sender: currentUsername, userId: currentUserId, text: textMessage, timestamp: firebase.firestore.FieldValue.serverTimestamp() }).then(() => { document.getElementById("message-input").value = ""; // تفريغ حقل الكتابة بعد الإرسال بنجاح }).catch((error) => { alert("فشل إرسال الرسالة، تأكد من اتصالك وقواعد الحماية."); }); } // 4. لوحة تحكم وصلاحيات المالك (الأدمن الرسمي) // دالة تعديل الخلفية (يستدعيها الملك لتتغير فوراً عند الجميع) function updateRoomBackground(newBackgroundUrl) { db.collection("rooms").doc(DEFAULT_ROOM).update({ background: newBackgroundUrl }).then(() => { console.log("تم تحديث الخلفية بنجاح في قاعدة البيانات وتعميمها على الكل."); }).catch((error) => { console.error("لم يتم التعديل، تأكد من صلاحيات الأدمن:", error); }); } // دالة إنشاء غرفة جديدة عند الضغط على الزر المخفي الخاص بصاحب الموقع function createNewRoomByOwner(roomName) { if (!roomName || roomName.trim() === "") return; const uniqueRoomId = "room_" + Date.now(); // توليد معرّف فريد غير متكرر بناءً على الوقت db.collection("rooms").doc(uniqueRoomId).set({ name: roomName, createdBy: "owner", createdAt: firebase.firestore.FieldValue.serverTimestamp() }).then(() => { alert("تم إنشاء الغرفة الجديدة بنجاح وستظهر عند جميع الأعضاء فوراً."); }).catch((error) => { console.error("خطأ في إنشاء الغرفة الفرعية:", error); }); } // 5. تشغيل واكتشاف أحداث أزرار الواجهة (HTML Event Listeners) document.addEventListener("DOMContentLoaded", () => { // تشغيل المستمعات اللحظية فور تحميل الصفحة بالكامل لضمان استقرار البث listenToChatUpdates(); // ربط زر الإرسال التقليدي في شاشتك بالدالة البرمجية const sendBtn = document.getElementById("send-btn"); // تأكد أن ID زر الإرسال عندك هو send-btn const messageInput = document.getElementById("message-input"); // تأكد أن ID حقل النص هو message-input if (sendBtn && messageInput) { sendBtn.addEventListener("click", () => { sendMessage(messageInput.value); }); // دعم الإرسال عند الضغط على زر Enter من لوحة المفاتيح messageInput.addEventListener("keypress", (e) => { if (e.key === "Enter") { sendMessage(messageInput.value); } }); } });
